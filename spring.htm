Repo-sizing grep commands (Instancio fixture-viability check)

1. Total types (denominator):
grep -rEl "(class|interface|enum|record)[[:space:]]+[A-Za-z_]" --include=*.java . | wc -l

2. Interfaces / abstract classes:
grep -rEl "interface[[:space:]]+[A-Za-z_]" --include=*.java . | wc -l
grep -rEl "abstract[[:space:]]+class[[:space:]]+[A-Za-z_]" --include=*.java . | wc -l

3. Distinct interface/abstract type names (the real cost number):
grep -rhoE "interface[[:space:]]+[A-Za-z_][A-Za-z0-9_]*" --include=*.java . | awk '{print $NF}' > iface_types.txt
grep -rhoE "abstract[[:space:]]+class[[:space:]]+[A-Za-z_][A-Za-z0-9_]*" --include=*.java . | awk '{print $NF}' >> iface_types.txt
sort -u iface_types.txt -o iface_types.txt
wc -l iface_types.txt

4. Exposure (classes with a field of one of those types):
while read T; do
grep -rlE "private[[:space:]]+(final[[:space:]]+)?$T[[:space:]]+[A-Za-z_]" --include=*.java .
done < iface_types.txt | sort -u | wc -l

5. Generic class declarations:
grep -rEl "class[[:space:]]+[A-Za-z_][A-Za-z0-9_]*[[:space:]]*<" --include=*.java . | wc -l

# testgen (drop-in)

Copy this folder into the **SUT module root** as `testgen/`. No foreign classpath — tools load SUT types via the module’s own test classpath (`Class.forName`).

## Setup

1. Copy this folder to `<module-root>/testgen/`
2. In IntelliJ: mark `testgen/` as **Test Sources Root**
3. Set Run configuration **Working directory** to the **module root** (parent of `testgen/`)
4. Ensure the module test classpath has: **javapoet**, **javaparser**, **instancio**, **mockito**, **junit** (Jupiter)
5. Edit `testgen/classes.txt` (one FQCN per line; `#` comments OK)
6. Run `com.testgen.Runner`

Comment out `scan` / `inject` / `generate` in `Runner.main` to skip a step. Set `Config.STRIP = true` and run only inject to remove markers.

## I/O

| Item | Path |
|------|------|
| Classes list | `MODULE/testgen/classes.txt` |
| Dependency graphs | `MODULE/testgen/reports/{Fqcn}.md` |
| Main sources (inject) | `MODULE/src/main/java` (override `Config.MAIN_SOURCES`) |
| Test skeletons | `MODULE/src/test/java` (override `Config.TEST_SOURCES`) |

`MODULE` = process working directory (must be the SUT module root).

| Step | Effect |
|------|--------|
| **scan** | Writes graphs under `testgen/reports/` |
| **inject** | Writes `// BEGIN testgen-dependencies` … blocks into production sources |
| **generate** | Writes `*{TEST_NAME_SUFFIX}` skeletons under test sources (skips if file exists) |

## Notes

- Generated skeletons are a starting point; they may need fixes before they compile/pass.
- Do not keep a Gradle wrapper inside this drop-in folder — dependencies come from the host module.

com/config/Config.java

package com.config;

import java.nio.file.Files;
import java.nio.file.Path;
import java.util.List;

/**
 * Drop-in contract: Run working directory = SUT module root.
 * This folder lives at {@code MODULE/testgen/}.
 */
public final class Config {

    /** SUT module root = process working directory. */
    public static final Path MODULE = Path.of("").toAbsolutePath().normalize();

    /**
     * Production source root.
     * null → {@code MODULE/src/main/java}
     */
    public static final Path MAIN_SOURCES = null;

    /**
     * Test source root (skeletons written here).
     * null → {@code MODULE/src/test/java}
     */
    public static final Path TEST_SOURCES = null;

    public static final String TEST_NAME_SUFFIX = "Test";

    /**
     * {@code true}: inject step removes dependency-graph markers from source (no inject).
     * {@code false}: inject step writes dependency-graph markers into source.
     */
    public static final boolean STRIP = false;

    private Config() {}

    /** {@code MODULE/testgen/classes.txt} */
    public static Path classesList() {
        return MODULE.resolve("testgen").resolve("classes.txt");
    }

    /** {@code MODULE/testgen/reports} */
    public static Path reportsDir() {
        return MODULE.resolve("testgen").resolve("reports");
    }

    public static Path mainSourceRoot() {
        return MAIN_SOURCES != null
                ? MAIN_SOURCES.toAbsolutePath().normalize()
                : MODULE.resolve("src").resolve("main").resolve("java");
    }

    public static Path testSourceRoot() {
        return TEST_SOURCES != null
                ? TEST_SOURCES.toAbsolutePath().normalize()
                : MODULE.resolve("src").resolve("test").resolve("java");
    }

    public static List<String> readClasses() throws Exception {
        Path list = classesList();
        if (!Files.isRegularFile(list)) {
            throw new IllegalStateException("classes list not found: " + list);
        }
        return Files.readAllLines(list).stream()
                .map(String::trim)
                .filter(line -> !line.isEmpty() && !line.startsWith("#"))
                .toList();
    }
}

com/testgen/DepGraphScanner.java

package com.testgen;

import com.config.Config;

import com.github.javaparser.StaticJavaParser;
import com.github.javaparser.ast.CompilationUnit;
import com.github.javaparser.ast.body.ClassOrInterfaceDeclaration;
import com.github.javaparser.ast.body.MethodDeclaration;
import com.github.javaparser.ast.expr.MethodCallExpr;
import com.github.javaparser.ast.expr.ObjectCreationExpr;
import com.github.javaparser.ast.visitor.VoidVisitorAdapter;

import java.lang.reflect.Constructor;
import java.lang.reflect.Field;
import java.lang.reflect.Method;
import java.lang.reflect.Modifier;
import java.lang.reflect.Parameter;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.Comparator;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Optional;
import java.util.Set;
import java.util.stream.Collectors;

/**
 * Step 1: scan SUTs → write per-class dependency graphs (no source injection).
 *
 * Output: {@code MODULE/testgen/reports/{Fqcn}.md}
 */
public final class DepGraphScanner {

    /** Cap call/new sites kept per method for LLM-sized comments. */
    static final int MAX_USES_PER_METHOD = 20;

    private DepGraphScanner() {}

    public static void run() throws Exception {
        Path outDir = Config.reportsDir();
        Files.createDirectories(outDir);
        List<String> classes = Config.readClasses();
        Path mainSourceRoot = Config.mainSourceRoot();

        int wrote = 0;
        for (String fqcn : classes) {
            Class<?> sut = Class.forName(fqcn);
            String markdown = scan(sut, mainSourceRoot);
            Path out = outDir.resolve(fqcn + ".md");
            Files.writeString(out, markdown);
            System.out.println("Wrote " + out);
            wrote++;
        }
        System.out.println("Done. " + wrote + " graph(s) in " + outDir.toAbsolutePath());
    }

    static String scan(Class<?> sut, Path mainSourceRoot) throws Exception {
        StringBuilder sb = new StringBuilder();
        sb.append("# testgen-dependencies\n");
        sb.append("sut: ").append(sut.getName()).append('\n');
        sb.append("hop: 1\n");
        sb.append("staticOnly: ").append(isStaticOnly(sut)).append('\n');
        sb.append('\n');

        sb.append("## fields\n");
        for (Field f : instanceAndStaticFields(sut)) {
            boolean iface = f.getType().isInterface();
            String kind = Modifier.isStatic(f.getModifiers()) ? "static" : "instance";
            String tip = iface ? "iface → mock" : "concrete → Instancio/null-risk";
            sb.append("- ").append(f.getName()).append(": ").append(typeLabel(f.getType()))
                    .append("  [").append(kind).append("; ").append(tip).append("]\n");
        }
        if (instanceAndStaticFields(sut).isEmpty()) {
            sb.append("- (none)\n");
        }
        sb.append('\n');

        sb.append("## constructors\n");
        Constructor<?>[] ctors = sut.getDeclaredConstructors();
        Arrays.sort(ctors, Comparator.comparingInt((Constructor<?> c) -> c.getParameterCount()).reversed());
        if (ctors.length == 0) {
            sb.append("- (none)\n");
        } else {
            for (Constructor<?> c : ctors) {
                if (!Modifier.isPublic(c.getModifiers()) && !Modifier.isProtected(c.getModifiers())) {
                    continue;
                }
                sb.append("- ").append(ctorSig(c)).append('\n');
            }
        }
        sb.append('\n');

        sb.append("## methods\n");
        Optional<CompilationUnit> cu = parseSource(sut, mainSourceRoot);
        List<Method> methods = publicMethods(sut);
        if (methods.isEmpty()) {
            sb.append("- (none)\n");
        }
        for (Method m : methods) {
            sb.append("### ").append(methodKey(m)).append('\n');
            sb.append("- static: ").append(Modifier.isStatic(m.getModifiers())).append('\n');
            sb.append("- returns: ").append(typeLabel(m.getReturnType())).append('\n');
            Class<?>[] checked = m.getExceptionTypes();
            if (checked.length > 0) {
                sb.append("- throwsChecked: ")
                        .append(Arrays.stream(checked).map(DepGraphScanner::typeLabel)
                                .collect(Collectors.joining(", ")))
                        .append('\n');
            } else {
                sb.append("- throwsChecked: (none)\n");
            }
            sb.append("- params:\n");
            if (m.getParameterCount() == 0) {
                sb.append("  - (none)\n");
            } else {
                for (Parameter p : m.getParameters()) {
                    boolean iface = p.getType().isInterface();
                    sb.append("  - ").append(p.getName()).append(": ").append(typeLabel(p.getType()))
                            .append(iface ? "  [iface → mock]\n" : "\n");
                }
            }

            List<String> uses = cu.map(unit -> hop1Uses(unit, sut.getSimpleName(), m))
                    .orElse(List.of("(source not found — inventory only)"));
            sb.append("- uses (hop 1):\n");
            if (uses.isEmpty()) {
                sb.append("  - (none found)\n");
            } else {
                for (String u : uses) {
                    sb.append("  - ").append(u).append('\n');
                }
            }
            sb.append('\n');
        }

        sb.append("## notes\n");
        sb.append("- hop 1 = direct calls / new in method body (AST); not transitive.\n");
        sb.append("- Logger / static init often ignored by skeleton; stub or exclude if NPE.\n");
        sb.append("- Unchecked exceptions (NPE, IAE) are not in throwsChecked — Duo/error loop.\n");
        return sb.toString();
    }

    private static Optional<CompilationUnit> parseSource(Class<?> sut, Path mainSourceRoot) {
        Path src = mainSourceRoot
                .resolve(sut.getPackageName().replace('.', '/'))
                .resolve(sut.getSimpleName() + ".java");
        if (!Files.isRegularFile(src)) {
            return Optional.empty();
        }
        try {
            return Optional.of(StaticJavaParser.parse(src));
        } catch (Exception e) {
            System.err.println("WARN parse failed " + src + ": " + e.getMessage());
            return Optional.empty();
        }
    }

    private static List<String> hop1Uses(CompilationUnit cu, String simpleName, Method method) {
        Optional<MethodDeclaration> md = findMethod(cu, simpleName, method);
        if (md.isEmpty()) {
            return List.of("(method body not located in source)");
        }
        LinkedHashSet<String> uses = new LinkedHashSet<>();
        md.get().accept(new VoidVisitorAdapter<Void>() {
            @Override
            public void visit(MethodCallExpr n, Void arg) {
                super.visit(n, arg);
                String name = n.getNameAsString();
                if (isNoise(name)) {
                    return;
                }
                String scope = n.getScope().map(Object::toString).orElse("this");
                if (isNoiseScope(scope)) {
                    return;
                }
                uses.add(scope + "." + name + "(" + n.getArguments().size() + ")");
            }

            @Override
            public void visit(ObjectCreationExpr n, Void arg) {
                super.visit(n, arg);
                uses.add("new " + n.getType().asString() + "(" + n.getArguments().size() + ")");
            }
        }, null);

        List<String> out = new ArrayList<>(uses);
        if (out.size() > MAX_USES_PER_METHOD) {
            return new ArrayList<>(out.subList(0, MAX_USES_PER_METHOD));
        }
        return out;
    }

    private static Optional<MethodDeclaration> findMethod(CompilationUnit cu, String simpleName, Method method) {
        Optional<ClassOrInterfaceDeclaration> type = cu.getClassByName(simpleName);
        if (type.isEmpty()) {
            return Optional.empty();
        }
        List<MethodDeclaration> candidates = type.get().getMethodsByName(method.getName());
        for (MethodDeclaration md : candidates) {
            if (md.getParameters().size() != method.getParameterCount()) {
                continue;
            }
            if (Modifier.isStatic(method.getModifiers()) != md.isStatic()) {
                continue;
            }
            return Optional.of(md);
        }
        return candidates.stream().findFirst();
    }

    private static boolean isNoise(String methodName) {
        return Set.of("equals", "hashCode", "toString", "getClass", "valueOf",
                "requireNonNull", "requireNonBlank").contains(methodName);
    }

    private static boolean isNoiseScope(String scope) {
        return scope.startsWith("Objects")
                || scope.startsWith("System")
                || scope.startsWith("Ids")
                || scope.equals("Math");
    }

    private static List<Field> instanceAndStaticFields(Class<?> sut) {
        List<Field> out = new ArrayList<>();
        Class<?> cur = sut;
        while (cur != null && cur != Object.class) {
            for (Field f : cur.getDeclaredFields()) {
                if (f.isSynthetic()) {
                    continue;
                }
                out.add(f);
            }
            cur = cur.getSuperclass();
        }
        return out;
    }

    private static List<Method> publicMethods(Class<?> sut) {
        List<Method> out = new ArrayList<>();
        for (Method m : sut.getDeclaredMethods()) {
            if (!Modifier.isPublic(m.getModifiers()) || m.isSynthetic() || m.isBridge()) {
                continue;
            }
            String n = m.getName();
            if (n.equals("equals") || n.equals("hashCode") || n.equals("toString")) {
                continue;
            }
            out.add(m);
        }
        if (filterInstance(sut.getDeclaredMethods()).isEmpty()
                || onlyAccessors(filterInstance(sut.getDeclaredMethods()))) {
            Class<?> parent = sut.getSuperclass();
            if (parent != null && parent != Object.class) {
                for (Method m : parent.getDeclaredMethods()) {
                    if (Modifier.isPublic(m.getModifiers()) && !Modifier.isStatic(m.getModifiers())
                            && !m.isSynthetic() && !m.isBridge()) {
                        out.add(m);
                    }
                }
            }
        }
        LinkedHashSet<String> seen = new LinkedHashSet<>();
        List<Method> deduped = new ArrayList<>();
        for (Method m : out) {
            String key = methodKey(m);
            if (seen.add(key)) {
                deduped.add(m);
            }
        }
        deduped.sort(Comparator.comparing(DepGraphScanner::methodKey));
        return deduped;
    }

    private static List<Method> filterInstance(Method[] methods) {
        List<Method> out = new ArrayList<>();
        for (Method m : methods) {
            if (Modifier.isPublic(m.getModifiers()) && !Modifier.isStatic(m.getModifiers())
                    && !m.isSynthetic() && !m.isBridge()) {
                out.add(m);
            }
        }
        return out;
    }

    private static boolean onlyAccessors(List<Method> methods) {
        if (methods.isEmpty()) {
            return true;
        }
        for (Method m : methods) {
            if (m.getParameterCount() != 0 || m.getReturnType() == void.class) {
                return false;
            }
            String n = m.getName();
            if (!n.startsWith("get") && !n.startsWith("is")) {
                return false;
            }
        }
        return true;
    }

    private static boolean isStaticOnly(Class<?> sut) {
        boolean anyInstance = false;
        boolean anyStatic = false;
        for (Method m : sut.getDeclaredMethods()) {
            if (!Modifier.isPublic(m.getModifiers()) || m.isSynthetic() || m.isBridge()) {
                continue;
            }
            if (Modifier.isStatic(m.getModifiers())) {
                anyStatic = true;
            } else {
                anyInstance = true;
            }
        }
        return !anyInstance && anyStatic;
    }

    private static String methodKey(Method m) {
        String params = Arrays.stream(m.getParameterTypes())
                .map(DepGraphScanner::typeLabel)
                .collect(Collectors.joining(", "));
        return m.getName() + "(" + params + ")";
    }

    private static String ctorSig(Constructor<?> c) {
        String params = Arrays.stream(c.getParameterTypes())
                .map(DepGraphScanner::typeLabel)
                .collect(Collectors.joining(", "));
        return c.getDeclaringClass().getSimpleName() + "(" + params + ")";
    }

    private static String typeLabel(Class<?> type) {
        if (type.isArray()) {
            return typeLabel(type.getComponentType()) + "[]";
        }
        return type.getName();
    }
}

com/testgen/DepGraphInjector.java

package com.testgen;

import com.config.Config;

import com.github.javaparser.StaticJavaParser;
import com.github.javaparser.ast.CompilationUnit;
import com.github.javaparser.ast.body.ClassOrInterfaceDeclaration;
import com.github.javaparser.ast.body.TypeDeclaration;

import java.nio.file.Files;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

/**
 * Step 2: inject per-SUT dependency graphs into source as line comments under the class brace.
 *
 * Markers (idempotent replace):
 *   // BEGIN testgen-dependencies
 *   ...
 *   // END testgen-dependencies
 *
 * Also strips legacy // BEGIN testgen-deps … // END testgen-deps blocks.
 */
public final class DepGraphInjector {

    static final String BEGIN = "// BEGIN testgen-dependencies";
    static final String END = "// END testgen-dependencies";

    /** Legacy short name — still stripped on inject/strip so older SUTs clean up. */
    static final String LEGACY_BEGIN = "// BEGIN testgen-deps";
    static final String LEGACY_END = "// END testgen-deps";

    private static final Pattern MARKER_BLOCK = Pattern.compile(
            "(?m)^[ \\t]*" + Pattern.quote(BEGIN) + "[\\s\\S]*?"
                    + "^[ \\t]*" + Pattern.quote(END) + "[ \\t]*\\R?");

    private static final Pattern LEGACY_MARKER_BLOCK = Pattern.compile(
            "(?m)^[ \\t]*" + Pattern.quote(LEGACY_BEGIN) + "[\\s\\S]*?"
                    + "^[ \\t]*" + Pattern.quote(LEGACY_END) + "[ \\t]*\\R?");

    private DepGraphInjector() {}

    public static void run(boolean strip) throws Exception {
        List<String> classes = Config.readClasses();
        Path mainSourceRoot = Config.mainSourceRoot();
        Path dependenciesDir = Config.reportsDir();

        int ok = 0;
        for (String fqcn : classes) {
            int lastDot = fqcn.lastIndexOf('.');
            Path src = mainSourceRoot
                    .resolve(fqcn.substring(0, lastDot).replace('.', '/'))
                    .resolve(fqcn.substring(lastDot + 1) + ".java");
            if (!Files.isRegularFile(src)) {
                System.err.println("SKIP missing source: " + src);
                continue;
            }
            if (strip) {
                stripFile(src);
                System.out.println("Stripped " + src);
            } else {
                Path graph = dependenciesDir.resolve(fqcn + ".md");
                if (!Files.isRegularFile(graph)) {
                    System.err.println("SKIP missing graph (run scan first): " + graph);
                    continue;
                }
                injectFile(src, fqcn, Files.readString(graph));
                System.out.println("Injected " + src);
            }
            ok++;
        }
        System.out.println((strip ? "Strip" : "Inject") + " done for " + ok + " file(s).");
    }

    static void injectFile(Path src, String fqcn, String markdown) throws Exception {
        String text = Files.readString(src);
        text = stripAllMarkers(text);
        text = trimTrailingBlankLines(text);

        String commentBlock = toCommentBlock(markdown);
        String simpleName = fqcn.substring(fqcn.lastIndexOf('.') + 1);
        String updated = insertAfterClassClosingBrace(text, simpleName, commentBlock);
        Files.writeString(src, updated);
    }

    static void stripFile(Path src) throws Exception {
        String text = Files.readString(src);
        String stripped = stripAllMarkers(text);
        stripped = trimTrailingBlankLines(stripped);
        if (!stripped.endsWith("\n")) {
            stripped = stripped + "\n";
        }
        Files.writeString(src, stripped);
    }

    static String stripAllMarkers(String text) {
        text = MARKER_BLOCK.matcher(text).replaceAll("");
        text = LEGACY_MARKER_BLOCK.matcher(text).replaceAll("");
        return text;
    }

    /** Each markdown line becomes a // comment line; wrapped by BEGIN/END markers. */
    static String toCommentBlock(String markdown) {
        StringBuilder sb = new StringBuilder();
        sb.append(BEGIN).append('\n');
        for (String line : markdown.replace("\r\n", "\n").replace('\r', '\n').split("\n", -1)) {
            if (line.isEmpty()) {
                sb.append("//\n");
            } else {
                sb.append("// ").append(line).append('\n');
            }
        }
        sb.append(END).append('\n');
        return sb.toString();
    }

    static String insertAfterClassClosingBrace(String source, String simpleName, String commentBlock)
            throws Exception {
        CompilationUnit cu = StaticJavaParser.parse(source);
        ClassOrInterfaceDeclaration type = cu.getClassByName(simpleName)
                .or(() -> findType(cu, simpleName))
                .orElseThrow(() -> new IllegalStateException("Type not found: " + simpleName));

        int endLine = type.getEnd()
                .orElseThrow(() -> new IllegalStateException("No end position for " + simpleName))
                .line; // 1-based

        List<String> lines = new ArrayList<>(List.of(source.replace("\r\n", "\n").split("\n", -1)));
        int insertAt = Math.min(endLine, lines.size());
        List<String> commentLines = List.of(commentBlock.replace("\r\n", "\n").split("\n", -1));
        if (!commentLines.isEmpty() && commentLines.get(commentLines.size() - 1).isEmpty()) {
            commentLines = commentLines.subList(0, commentLines.size() - 1);
        }

        List<String> out = new ArrayList<>();
        out.addAll(lines.subList(0, insertAt));
        if (!out.isEmpty() && !out.get(out.size() - 1).isBlank()) {
            out.add("");
        }
        out.addAll(commentLines);
        if (insertAt < lines.size()) {
            out.addAll(lines.subList(insertAt, lines.size()));
        }
        String joined = String.join("\n", out);
        if (!joined.endsWith("\n")) {
            joined = joined + "\n";
        }
        return joined;
    }

    private static java.util.Optional<ClassOrInterfaceDeclaration> findType(CompilationUnit cu, String simpleName) {
        for (TypeDeclaration<?> t : cu.getTypes()) {
            if (t instanceof ClassOrInterfaceDeclaration cid && cid.getNameAsString().equals(simpleName)) {
                return java.util.Optional.of(cid);
            }
        }
        return java.util.Optional.empty();
    }

    private static String trimTrailingBlankLines(String text) {
        text = text.replace("\r\n", "\n").replace('\r', '\n');
        while (text.endsWith("\n\n")) {
            text = text.substring(0, text.length() - 1);
        }
        Matcher leftover = MARKER_BLOCK.matcher(text);
        if (leftover.find()) {
            text = leftover.replaceAll("");
        }
        leftover = LEGACY_MARKER_BLOCK.matcher(text);
        if (leftover.find()) {
            text = leftover.replaceAll("");
        }
        return text;
    }
}

package com.testgen;

import com.config.Config;

import com.squareup.javapoet.ClassName;
import com.squareup.javapoet.CodeBlock;
import com.squareup.javapoet.FieldSpec;
import com.squareup.javapoet.JavaFile;
import com.squareup.javapoet.MethodSpec;
import com.squareup.javapoet.TypeName;
import com.squareup.javapoet.TypeSpec;

import javax.lang.model.element.Modifier;
import java.lang.reflect.Field;
import java.lang.reflect.Method;
import java.lang.reflect.Parameter;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.HashSet;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Set;

/**
 * Batch skeleton generator:
 * - Instancio for concrete types + SUT instance (not manual ctor wiring)
 * - Mockito.mock for interfaces; minimal stubbing for smoke act
 * - instance + static in same class; static-only SUTs get static tests only
 * - SUT contract methods: declared; if empty/thin → interfaces + immediate parent
 * - mirrors declared checked exceptions on generated @Test methods
 * - skip if *{suffix} already exists
 *
 * Array TypeName via {@link TypeName#get(java.lang.reflect.Type)} (javaPoetType).
 */
public final class TestSkeletonGenerator {

    private static final ClassName INSTANCIO = ClassName.get("org.instancio", "Instancio");
    private static final ClassName SELECT = ClassName.get("org.instancio", "Select");
    private static final ClassName MOCKITO = ClassName.get("org.mockito", "Mockito");
    private static final ClassName MATCHERS = ClassName.get("org.mockito", "ArgumentMatchers");
    private static final ClassName TEST = ClassName.get("org.junit.jupiter.api", "Test");
    private static final ClassName BEFORE_EACH = ClassName.get("org.junit.jupiter.api", "BeforeEach");
    private static final ClassName ASSERTIONS = ClassName.get("org.junit.jupiter.api", "Assertions");

    private TestSkeletonGenerator() {}

    public static void run() throws Exception {
        Path testSourceRoot = Config.testSourceRoot();
        String suffix = Config.TEST_NAME_SUFFIX;
        List<String> classes = Config.readClasses();

        for (String fqcn : classes) {
            writeSkeleton(Class.forName(fqcn), testSourceRoot, suffix);
        }
    }

    static void writeSkeleton(Class<?> sut, Path testSourceRoot, String suffix) throws Exception {
        String testPackage = sut.getPackageName();
        String testSimpleName = sut.getSimpleName() + suffix;
        Path target = testSourceRoot
                .resolve(testPackage.replace('.', '/'))
                .resolve(testSimpleName + ".java");
        if (Files.exists(target)) {
            System.out.println("SKIP (exists): " + target);
            return;
        }

        boolean staticOnly = isStaticOnlyClass(sut);
        List<Method> instanceMethods = staticOnly ? List.of() : contractInstanceMethods(sut);
        List<Method> staticMethods = contractStaticMethods(sut);
        boolean staticPrefix = !staticOnly && !instanceMethods.isEmpty();

        TypeSpec.Builder testClass = TypeSpec.classBuilder(testSimpleName)
                .addModifiers(Modifier.PUBLIC)
                .addJavadoc(
                        "AUTO-GENERATED arrange skeleton for $L.\n"
                                + "Instancio SUT + locals per test; Duo fills real asserts later.\n",
                        sut.getName());

        if (!instanceMethods.isEmpty()) {
            testClass.addField(FieldSpec.builder(ClassName.get(sut), "sut", Modifier.PRIVATE).build());
            testClass.addMethod(setUpMethod(sut));
            Set<String> usedNames = new HashSet<>();
            for (Method m : instanceMethods) {
                testClass.addMethod(instanceTest(m, usedNames));
            }
        }
        if (!staticMethods.isEmpty()) {
            Set<String> usedNames = instanceMethods.stream()
                    .map(TestSkeletonGenerator::safeName)
                    .collect(java.util.stream.Collectors.toCollection(HashSet::new));
            for (Method m : staticMethods) {
                testClass.addMethod(staticTest(sut, m, staticPrefix, usedNames));
            }
        }
        if (instanceMethods.isEmpty() && staticMethods.isEmpty()) {
            System.out.println("WARN no public methods: " + sut.getName());
            testClass.addJavadoc("No public methods discovered.\n");
        }

        JavaFile.builder(testPackage, testClass.build())
                .addFileComment("AUTO-GENERATED by testgen TestSkeletonGenerator")
                .build()
                .writeTo(testSourceRoot);

        System.out.println("Wrote " + testPackage + "." + testSimpleName
                + " (instance=" + instanceMethods.size() + ", static=" + staticMethods.size() + ")");
    }

    private static MethodSpec setUpMethod(Class<?> sut) {
        MethodSpec.Builder b = MethodSpec.methodBuilder("setUp")
                .addAnnotation(BEFORE_EACH)
                .addModifiers(Modifier.PUBLIC);

        List<FieldMockRef> fieldMocks = new ArrayList<>();
        for (Field field : instanceFields(sut)) {
            Class<?> ft = field.getType();
            if (!ft.isInterface()) {
                continue;
            }
            String var = field.getName() + "Mock";
            fieldMocks.add(new FieldMockRef(ft, var, field.getDeclaringClass(), field.getName()));
            b.addStatement("$T $L = $T.mock($T.class)",
                    ClassName.get(ft), var, MOCKITO, ClassName.get(ft));
        }

        if (fieldMocks.isEmpty()) {
            b.addStatement("sut = $T.create($T.class)", INSTANCIO, ClassName.get(sut));
        } else {
            CodeBlock.Builder chain = CodeBlock.builder().add("sut = $T.of($T.class)",
                    INSTANCIO, ClassName.get(sut));
            for (FieldMockRef mock : fieldMocks) {
                chain.add("\n    .set($T.field($T.class, $S), $L)",
                        SELECT, ClassName.get(mock.declaringClass), mock.fieldName, mock.varName);
            }
            chain.add("\n    .create();\n");
            b.addCode(chain.build());
        }

        for (FieldMockRef mock : fieldMocks) {
            b.addCode(minimalStubs(new MockRef(mock.type(), mock.varName())));
        }
        return b.build();
    }

    private static MethodSpec instanceTest(Method method, Set<String> usedNames) {
        MethodSpec.Builder b = MethodSpec.methodBuilder(uniqueName(safeName(method), usedNames))
                .addModifiers(Modifier.PUBLIC)
                .addAnnotation(TEST);
        addDeclaredExceptions(method, b);

        List<MockRef> mocks = new ArrayList<>();
        List<String> methodArgNames = new ArrayList<>();
        int j = 0;
        for (Parameter p : method.getParameters()) {
            String var = "a" + (j++) + "_" + simpleParamName(p);
            b.addCode(arrangeLine(p.getType(), var, mocks));
            methodArgNames.add(var);
        }

        for (MockRef mock : mocks) {
            b.addCode(minimalStubs(mock));
        }

        String callArgs = String.join(", ", methodArgNames);
        if (method.getReturnType() == void.class) {
            b.addStatement("sut.$L($L)", method.getName(), callArgs);
            b.addStatement("$T.assertNotNull(sut)", ASSERTIONS);
        } else if (method.getReturnType().isPrimitive()) {
            b.addStatement("sut.$L($L)", method.getName(), callArgs);
            b.addStatement("$T.assertTrue(true)", ASSERTIONS);
        } else {
            b.addStatement("$T result = sut.$L($L)",
                    TypeName.get(method.getReturnType()), method.getName(), callArgs);
            b.addStatement("$T.assertNotNull(result)", ASSERTIONS);
        }
        return b.build();
    }

    private static MethodSpec staticTest(Class<?> sut, Method method, boolean prefix, Set<String> usedNames) {
        String base = prefix ? "static_" + safeName(method) : safeName(method);
        MethodSpec.Builder b = MethodSpec.methodBuilder(uniqueName(base, usedNames))
                .addModifiers(Modifier.PUBLIC)
                .addAnnotation(TEST);
        addDeclaredExceptions(method, b);

        List<MockRef> mocks = new ArrayList<>();
        List<String> methodArgNames = new ArrayList<>();
        int j = 0;
        for (Parameter p : method.getParameters()) {
            String var = "a" + (j++) + "_" + simpleParamName(p);
            b.addCode(arrangeLine(p.getType(), var, mocks));
            methodArgNames.add(var);
        }
        for (MockRef mock : mocks) {
            b.addCode(minimalStubs(mock));
        }

        String callArgs = String.join(", ", methodArgNames);
        ClassName sutCn = ClassName.get(sut);
        if (method.getReturnType() == void.class) {
            b.addStatement("$T.$L($L)", sutCn, method.getName(), callArgs);
            b.addStatement("$T.assertTrue(true)", ASSERTIONS);
        } else if (method.getReturnType().isPrimitive()) {
            b.addStatement("$T.$L($L)", sutCn, method.getName(), callArgs);
            b.addStatement("$T.assertTrue(true)", ASSERTIONS);
        } else {
            b.addStatement("$T result = $T.$L($L)",
                    TypeName.get(method.getReturnType()), sutCn, method.getName(), callArgs);
            b.addStatement("$T.assertNotNull(result)", ASSERTIONS);
        }
        return b.build();
    }

    private static List<Field> instanceFields(Class<?> sut) {
        List<Field> fields = new ArrayList<>();
        Class<?> cur = sut;
        while (cur != null && cur != Object.class) {
            // Skip JDK frames (e.g. Throwable.suppressedExceptions) — not SUT collaborators.
            if (isJdkType(cur)) {
                cur = cur.getSuperclass();
                continue;
            }
            for (Field field : cur.getDeclaredFields()) {
                if (!java.lang.reflect.Modifier.isStatic(field.getModifiers())) {
                    fields.add(field);
                }
            }
            cur = cur.getSuperclass();
        }
        return fields;
    }

    private static boolean isJdkType(Class<?> type) {
        Package p = type.getPackage();
        if (p == null) {
            return false;
        }
        String n = p.getName();
        return n.equals("java") || n.startsWith("java.")
                || n.equals("javax") || n.startsWith("javax.")
                || n.equals("jdk") || n.startsWith("jdk.");
    }

    private static boolean isStaticOnlyClass(Class<?> sut) {
        return filterInstance(sut.getDeclaredMethods()).isEmpty()
                && !filterStatic(sut.getDeclaredMethods()).isEmpty();
    }

    /** Prefer TypeName.get so arrays (and generics) render correctly. */
    private static TypeName javaPoetType(Class<?> type) {
        return TypeName.get(type);
    }

    private static CodeBlock arrangeLine(Class<?> type, String varName, List<MockRef> mocks) {
        if (type.isInterface()) {
            mocks.add(new MockRef(type, varName));
            return CodeBlock.of("$T $L = $T.mock($T.class);\n",
                    ClassName.get(type), varName, MOCKITO, ClassName.get(type));
        }
        if (type == Class.class) {
            return CodeBlock.of("$T<?> $L = $T.class;\n",
                    ClassName.get(Class.class), varName, ClassName.get(String.class));
        }
        if (type == java.math.BigDecimal.class) {
            return CodeBlock.of("$T $L = $T.ZERO;\n",
                    ClassName.get(java.math.BigDecimal.class), varName,
                    ClassName.get(java.math.BigDecimal.class));
        }
        if (type.isPrimitive()) {
            Class<?> boxed = box(type);
            return CodeBlock.of("$T $L = $T.create($T.class);\n",
                    ClassName.get(boxed), varName, INSTANCIO, ClassName.get(boxed));
        }
        return arrangeConcrete(type, varName, mocks);
    }

    private static CodeBlock arrangeConcrete(Class<?> type, String varName, List<MockRef> mocks) {
        List<Field> ifaceFields = interfaceInstanceFields(type);
        if (ifaceFields.isEmpty()) {
            TypeName tn = javaPoetType(type);
            return CodeBlock.of("$T $L = $T.create($T.class);\n", tn, varName, INSTANCIO, tn);
        }
        CodeBlock.Builder block = CodeBlock.builder();
        for (Field field : ifaceFields) {
            String mockVar = varName + "_" + field.getName() + "Mock";
            mocks.add(new MockRef(field.getType(), mockVar));
            block.addStatement("$T $L = $T.mock($T.class)",
                    ClassName.get(field.getType()), mockVar, MOCKITO, ClassName.get(field.getType()));
        }
        CodeBlock.Builder chain = CodeBlock.builder().add("$T $L = $T.of($T.class)",
                ClassName.get(type), varName, INSTANCIO, ClassName.get(type));
        for (Field field : ifaceFields) {
            String mockVar = varName + "_" + field.getName() + "Mock";
            chain.add("\n    .set($T.field($T.class, $S), $L)",
                    SELECT, ClassName.get(field.getDeclaringClass()), field.getName(), mockVar);
        }
        chain.add("\n    .create();\n");
        block.add(chain.build());
        return block.build();
    }

    private static List<Field> interfaceInstanceFields(Class<?> type) {
        List<Field> out = new ArrayList<>();
        for (Field field : instanceFields(type)) {
            if (field.getType().isInterface()) {
                out.add(field);
            }
        }
        return out;
    }

    private static void addDeclaredExceptions(Method method, MethodSpec.Builder testMethod) {
        for (Class<?> ex : method.getExceptionTypes()) {
            testMethod.addException(TypeName.get(ex));
        }
    }

    private static CodeBlock minimalStubs(MockRef mock) {
        CodeBlock.Builder stubs = CodeBlock.builder();
        for (Method m : mock.type.getMethods()) {
            if (!include(m) || java.lang.reflect.Modifier.isStatic(m.getModifiers())) {
                continue;
            }
            CodeBlock anyArgs = anyMatchers(m.getParameterCount());
            Class<?> ret = m.getReturnType();
            if (ret == void.class) {
                stubs.addStatement("$T.doNothing().when($L).$L($L)",
                        MOCKITO, mock.varName, m.getName(), anyArgs);
            } else if (ret.isPrimitive()) {
                stubs.addStatement("$T.when($L.$L($L)).thenReturn($L)",
                        MOCKITO, mock.varName, m.getName(), anyArgs, primitiveDefaultExpr(ret));
            } else {
                TypeName retTn = javaPoetType(ret);
                stubs.addStatement("$T.when($L.$L($L)).thenReturn($T.create($T.class))",
                        MOCKITO, mock.varName, m.getName(), anyArgs, INSTANCIO, retTn);
            }
        }
        return stubs.build();
    }

    private static CodeBlock anyMatchers(int count) {
        if (count <= 0) {
            return CodeBlock.of("");
        }
        CodeBlock.Builder sb = CodeBlock.builder();
        for (int i = 0; i < count; i++) {
            if (i > 0) {
                sb.add(", ");
            }
            sb.add("$T.any()", MATCHERS);
        }
        return sb.build();
    }

    private static String primitiveDefaultExpr(Class<?> prim) {
        if (prim == boolean.class) return "false";
        if (prim == char.class) return "'\\0'";
        return "0";
    }

    private static List<Method> contractInstanceMethods(Class<?> sut) {
        List<Method> declared = filterInstance(sut.getDeclaredMethods());
        if (!isThin(declared)) {
            return dedupe(declared);
        }
        LinkedHashSet<Method> contract = new LinkedHashSet<>(declared);
        for (Class<?> iface : allInterfaces(sut)) {
            contract.addAll(filterInstance(iface.getMethods()));
        }
        Class<?> parent = sut.getSuperclass();
        if (parent != null && parent != Object.class) {
            contract.addAll(filterInstance(parent.getDeclaredMethods()));
        }
        return dedupe(new ArrayList<>(contract));
    }

    private static List<Method> contractStaticMethods(Class<?> sut) {
        List<Method> declared = filterStatic(sut.getDeclaredMethods());
        if (!declared.isEmpty()) {
            return dedupe(declared);
        }
        LinkedHashSet<Method> contract = new LinkedHashSet<>();
        for (Class<?> iface : allInterfaces(sut)) {
            contract.addAll(filterStatic(iface.getMethods()));
        }
        Class<?> parent = sut.getSuperclass();
        if (parent != null && parent != Object.class) {
            contract.addAll(filterStatic(parent.getDeclaredMethods()));
        }
        return dedupe(new ArrayList<>(contract));
    }

    private static boolean isThin(List<Method> declared) {
        if (declared.isEmpty()) {
            return true;
        }
        for (Method m : declared) {
            if (!looksLikeAccessor(m)) {
                return false;
            }
        }
        return true;
    }

    private static boolean looksLikeAccessor(Method m) {
        if (m.getParameterCount() != 0 || m.getReturnType() == void.class) {
            return false;
        }
        String n = m.getName();
        return n.startsWith("get") || n.startsWith("is");
    }

    private static List<Class<?>> allInterfaces(Class<?> type) {
        LinkedHashSet<Class<?>> out = new LinkedHashSet<>();
        Class<?> cur = type;
        while (cur != null && cur != Object.class) {
            out.addAll(Arrays.asList(cur.getInterfaces()));
            for (Class<?> iface : cur.getInterfaces()) {
                collectInterfaces(iface, out);
            }
            cur = cur.getSuperclass();
        }
        return new ArrayList<>(out);
    }

    private static void collectInterfaces(Class<?> iface, Set<Class<?>> out) {
        if (!out.add(iface)) {
            return;
        }
        for (Class<?> parent : iface.getInterfaces()) {
            collectInterfaces(parent, out);
        }
    }

    private static List<Method> filterInstance(Method[] methods) {
        List<Method> out = new ArrayList<>();
        for (Method m : methods) {
            if (include(m) && !java.lang.reflect.Modifier.isStatic(m.getModifiers())) {
                out.add(m);
            }
        }
        return out;
    }

    private static List<Method> filterStatic(Method[] methods) {
        List<Method> out = new ArrayList<>();
        for (Method m : methods) {
            if (include(m) && java.lang.reflect.Modifier.isStatic(m.getModifiers())) {
                out.add(m);
            }
        }
        return out;
    }

    private static List<Method> dedupe(List<Method> methods) {
        List<Method> out = new ArrayList<>();
        Set<String> seen = new HashSet<>();
        for (Method m : methods) {
            String key = m.getName() + Arrays.toString(m.getParameterTypes());
            if (seen.add(key)) {
                out.add(m);
            }
        }
        return out;
    }

    private static boolean include(Method m) {
        int mod = m.getModifiers();
        if (!java.lang.reflect.Modifier.isPublic(mod)) {
            return false;
        }
        if (m.isSynthetic() || m.isBridge()) {
            return false;
        }
        String n = m.getName();
        if (n.equals("equals") || n.equals("hashCode") || n.equals("toString") || n.equals("getClass")
                || n.equals("notify") || n.equals("notifyAll") || n.equals("wait")) {
            return false;
        }
        return true;
    }

    private static String safeName(Method m) {
        String base = m.getName();
        if (m.getParameterCount() == 0) {
            return base;
        }
        return base + "_" + m.getParameterCount() + "args";
    }

    private static String uniqueName(String base, Set<String> usedNames) {
        String name = base;
        int i = 2;
        while (!usedNames.add(name)) {
            name = base + "_" + i++;
        }
        return name;
    }

    private static String simpleParamName(Parameter p) {
        String n = p.getName();
        if (n == null || n.startsWith("arg")) {
            n = p.getType().getSimpleName();
        }
        return n.replaceAll("[^A-Za-z0-9_]", "");
    }

    private static Class<?> box(Class<?> primitive) {
        if (primitive == boolean.class) return Boolean.class;
        if (primitive == int.class) return Integer.class;
        if (primitive == long.class) return Long.class;
        if (primitive == double.class) return Double.class;
        if (primitive == float.class) return Float.class;
        if (primitive == short.class) return Short.class;
        if (primitive == byte.class) return Byte.class;
        if (primitive == char.class) return Character.class;
        return primitive;
    }

    private record MockRef(Class<?> type, String varName) {}

    private record FieldMockRef(Class<?> type, String varName, Class<?> declaringClass, String fieldName) {}
}
